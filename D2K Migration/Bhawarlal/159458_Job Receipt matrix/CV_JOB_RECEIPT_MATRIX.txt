DROP VIEW GINVIEW.CV_JOB_RECEIPT_MATRIX;

/* Formatted on 25-01-2022 15:36:01 (QP5 v5.294) */
CREATE OR REPLACE FORCE VIEW GINVIEW.CV_JOB_RECEIPT_MATRIX
(
   JRCCODE,
   JOB_ORDER_NO,
   JOB_ORDER_DATE,
   REFERENCE_NO,
   PLAN_NO,
   HSN_CODE,
   ITEM,
   UOM,
   ITEM_SIZE,
   QTY,
   JOB_RATE
)
   BEQUEATH DEFINER
AS
   SELECT JRCCODE,
          D.JOB_ORDER_NO,
          D.JOB_ORDER_DATE,
          jo.REFERENCE_NO,
          D.PLAN_NO,
          HSN_CODE,
             CATEGORY1
          || ' '
          || CATEGORY2
          || ' '
          || CATEGORY3
          || ' '
          || CATEGORY4
          || ' '
          || CATEGORY5
          || ' '
          || CATEGORY6
             ITEM,
          UOM,
          TRIM (CATEGORY4)  ITEM_SIZE,
          RECEIVED_QUANTITY QTY,
          JOB_RATE
     FROM GINVIEW.LV_PRD_JOB_RECEIPT_DETAIL D,
          GINVIEW.LV_ITEM                   I,
          GINVIEW.LV_PRD_JOB_ORDER_HEADER   jo
    WHERE D.ASSEMBLY_ICODE = I.CODE AND D.JOBCODE = jo.CODE(+);
