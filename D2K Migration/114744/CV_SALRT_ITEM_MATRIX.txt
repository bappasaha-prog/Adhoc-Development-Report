create or replace view ginview.CV_SALRT_ITEM_MATRIX
AS
select ginview.fnc_uk UK, rtcode, category1||' '||category2||' '||category3||' '||category5||' '||category6 ITEm, Returned_Rate Rate, category4 Item_Size, sum(returned_quantity) Qty, sum(item_gross_amount) over (partition by rtcode,category1||' '||category2||' '||category3||' '||category5||' '||category6 order by rtcode) Amount
from ginview.lv_salrt_item A, ginview.lv_item B
where A.ICODE = B.code
and rtcode = 588055
group by 
rtcode, category1||' '||category2||' '||category3||' '||category5||' '||category6 , Returned_Rate,category4,item_gross_amount
order by
rtcode, category1||' '||category2||' '||category3||' '||category5||' '||category6 , Returned_Rate,category4,item_gross_amount
